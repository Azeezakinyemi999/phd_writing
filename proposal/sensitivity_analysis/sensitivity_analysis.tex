  \subsection{Regression-Based Methods}
  \begin{itemize}
    \item Standardized Regression Coefficients (SRC)
    \item Pearson and Spearman correlation
    \item Limitations: $R^2$ and nonlinearity
  \end{itemize}

  \subsection{Variance-Based Methods (Sobol Indices)}
  \begin{itemize}
    \item Functional ANOVA decomposition
    \item First-order index $\Si$ (main effect)
    \item Total-order index $\STi$
    \item Higher-order interaction indices
  \end{itemize}

  \subsection{Density- and Moment-Independent Methods}
  \begin{itemize}
    \item Delta sensitivity measure ($\delta_i$)
    \item Kullback--Leibler divergence-based measures
    \item Transformation invariance property
  \end{itemize}

  \subsection{Value of Information-Based Methods}
  \begin{itemize}
    \item Partial Expected Value of Perfect Information (EVPPI)
    \item Decision sensitivity vs.\ value sensitivity
  \end{itemize}
